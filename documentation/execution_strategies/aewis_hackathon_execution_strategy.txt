def risk_scoring(df):
    df = df.copy()
    df['avg'] = df[['Quiz1','Quiz2','Quiz3']].mean(1)
    df['decline'] = (df['Quiz1']-df['Quiz3'])/df['Quiz1']*100 >= 15
    df['low_attend'] = df['Attendance'] < 75
    df['low_avg'] = df['avg'] < 40
    df['risk'] = df[['decline','low_attend','low_avg']].any(1).map({True:'HIGH',False:'LOW'})
    df['reason'] = df.apply(lambda row: ', '.join([c for c in ['Decline','Attend','Avg'] 
                      if row[f'{"low_" if "Attend" in c or "Avg" in c else ""}{c.lower()}']]),1)
    return df
```

def generate_demo_data(n=500):
    students = [f"S{i:03d}" for i in range(1,n+1)]
    subjects = np.random.choice(['Math','Physics','Chemistry','English'], n)
    # 20% HIGH risks, 30% edge cases, 50% normal
    quiz1 = np.random.normal(75,15,n).clip(0,100)
    quiz3 = np.where(np.random.rand(n)<0.2, quiz1*0.75, quiz1 + np.random.normal(0,8,n))
    attend = np.random.normal(82,12,n).clip(0,100)
    return pd.DataFrame({
        'StudentID':students, 'Subject':subjects,
        'Quiz1':quiz1, 'Quiz2':quiz1 + np.random.normal(0,5,n), 
        'Quiz3':quiz3, 'Attendance':attend
    })
```
